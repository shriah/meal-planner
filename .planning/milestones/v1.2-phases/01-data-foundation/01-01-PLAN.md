   ```typescript
   // Component type discriminator — DATA-01
   export type ComponentType = 'base' | 'curry' | 'subzi' | 'extra';

   // Tag string literal unions — DATA-05
   export type DietaryTag = 'veg' | 'non-veg' | 'vegan' | 'jain' | 'eggetarian';
   export type ProteinTag = 'fish' | 'chicken' | 'mutton' | 'egg' | 'paneer' | 'dal' | 'none';
   export type RegionalTag = 'south-indian' | 'north-indian' | 'coastal-konkan' | 'pan-indian';
   export type OccasionTag = 'everyday' | 'fasting' | 'festive' | 'weekend';
   ```

   Then define the shared base interface:
   ```typescript
   interface BaseComponentFields {
     id?: number;
     name: string;
     componentType: ComponentType;
     dietary_tags: DietaryTag[];
     protein_tag?: ProteinTag;
     regional_tags: RegionalTag[];
     occasion_tags: OccasionTag[];
     notes?: string;
     created_at: string;
   }
   ```

   Then define discriminated union members:
   - `BaseRecord` extends `BaseComponentFields` with `componentType: 'base'` and required `base_type: BaseType`
   - `CurryRecord` extends `BaseComponentFields` with `componentType: 'curry'` and optional `curry_category?: string`
   - `SubziRecord` extends `BaseComponentFields` with `componentType: 'subzi'` and optional `compatible_base_types?: BaseType[]`
   - `ExtraRecord` extends `BaseComponentFields` with `componentType: 'extra'`, required `extra_category: ExtraCategory`, required `compatible_base_types: BaseType[]`, and optional `incompatible_curry_categories?: string[]`

   Then define:
   ```typescript
   export type MealComponent = BaseRecord | CurryRecord | SubziRecord | ExtraRecord;

   // Flat record type for Dexie storage — all fields merged, type-specific fields optional
   export type ComponentRecord = BaseComponentFields & {
     base_type?: BaseType;
     curry_category?: string;
     compatible_base_types?: BaseType[];
     extra_category?: ExtraCategory;
     incompatible_curry_categories?: string[];
   };
   ```

5. Create `src/db/client.ts` — Dexie v4 singleton with EntityTable:
   ```typescript
   import Dexie, { type EntityTable } from 'dexie';
   import type { ComponentRecord } from '@/types/component';
   import type { MealRecord, MealExtraRecord } from '@/types/meal';
   import type { UserPreferencesRecord } from '@/types/preferences';

   // Rule and SavedPlan stubs for future phases
   export interface RuleRecord {
     id?: number;
     text: string;
     compiled_filter?: unknown;
     is_active: boolean;
     created_at: string;
   }

   export interface SavedPlanRecord {
     id?: number;
     name: string;
     slots: unknown;
     created_at: string;
   }

   const db = new Dexie('FoodPlannerDB') as Dexie & {
     components: EntityTable<ComponentRecord, 'id'>;
     meals: EntityTable<MealRecord, 'id'>;
     meal_extras: EntityTable<MealExtraRecord, 'meal_id'>;
     rules: EntityTable<RuleRecord, 'id'>;
     saved_plans: EntityTable<SavedPlanRecord, 'id'>;
     preferences: EntityTable<UserPreferencesRecord, 'id'>;
   };

   db.version(1).stores({
     components: '++id, componentType, base_type, extra_category, *dietary_tags, *regional_tags, *occasion_tags',
     meals: '++id, base_id, curry_id, subzi_id',
     meal_extras: '[meal_id+component_id], meal_id, component_id',
     rules: '++id, is_active',
     saved_plans: '++id',
     preferences: 'id',
   });

   export { db };
   ```

6. Verify TypeScript compiles: run `npx tsc --noEmit` from project root.
